response (dict)
│
├─ messages (list)
│   ├─ HumanMessage
│   │    ├─ content: str                  # The user question
│   │    ├─ additional_kwargs: dict
│   │    ├─ response_metadata: dict
│   │    └─ id: str
│   │
│   ├─ AIMessage
│   │    ├─ content: str                  # Text generated by AI (can be empty if tool call)
│   │    ├─ additional_kwargs: dict
│   │    │    └─ tool_calls (list)
│   │    │         ├─ id: str
│   │    │         ├─ function (dict)
│   │    │         │    ├─ name: str (tool name)
│   │    │         │    └─ arguments: dict
│   │    │         └─ type: str ("function")
│   │    ├─ response_metadata: dict
│   │    │    ├─ token_usage: dict
│   │    │    ├─ model_name: str
│   │    │    ├─ finish_reason: str
│   │    │    └─ model_provider: str
│   │    ├─ id: str
│   │    ├─ tool_calls (list)             # Repeated tool calls extracted
│   │    └─ usage_metadata: dict
│   │
│   ├─ ToolMessage
│   │    ├─ content: str                  # Output of the tool
│   │    ├─ name: str                     # Tool name
│   │    ├─ id: str
│   │    └─ tool_call_id: str             # Maps to AIMessage tool call
│   │
│   └─ AIMessage (can repeat)             # Another AIMessage with final answer or SQL
│
